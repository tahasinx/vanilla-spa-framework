Quick Start

  1. Clone or download this framework.
  2. Run a local server (do not use file://).
  3. Open index.html in your browser.
  4. Define routes in routes/web.js and build features.
# Recommended local server command (Windows-friendly)
python -m http.server 5500 --bind 127.0.0.1

Open http://127.0.0.1:5500. Alternative: python -m http.server 5500 if localhost resolves correctly on your machine.

Routing (Unified + Laravel-like)

Define all web and API routes in routes/web.js using AppRouter.

// Named page routes
AppRouter.get('/', 'HomeController', 'index', { name: 'home' });
AppRouter.get('/docs', 'HomeController', 'docs', { name: 'docs' });

// Route groups with prefixes
AppRouter.group({ prefix: '/demo', namePrefix: 'demo.' }, () => {
  AppRouter.get('/', 'DemoController', 'index', { name: 'index' });
  AppRouter.get('/blog/{id}', 'DemoController', 'blogView', { name: 'blog.view' });
});

AppRouter.group({ prefix: '/api', namePrefix: 'api.' }, () => {
  AppRouter.get('/users', 'ApiController', 'getUsers', { name: 'users.index' });
  AppRouter.get('/inspire', 'ApiController', 'inspire', { name: 'inspire' });
});
// Named route helper
const blogPath = route('demo.blog.view', { id: 5 });
window.AppRouter.navigate(blogPath);

Tip: keep routing centralized in one file for predictable maintenance.

Middleware & Frontend Auth

This framework now supports route middleware for SPA navigation flow. It improves UX but does not replace backend authorization.

// Built-in middleware usage
AppRouter.get('/dashboard', 'DashboardController', 'index', {
  name: 'dashboard',
  middleware: ['auth']
});

AppRouter.get('/login', 'AuthController', 'login', {
  name: 'login',
  middleware: ['guest']
});
// Locale middleware (auto-detect locale + load translations)
AppRouter.get('/', 'HomeController', 'index', {
  name: 'home',
  middleware: ['locale']
});

// Optional locale from query/hash: /#/docs?lang=ar
// middleware/TestMiddleware.js
window.ProfileCompleteMiddleware = () => {
  const isComplete = Boolean(localStorage.getItem('profile.complete'));
  return isComplete ? true : '/';
};
// Middleware file (example)
// middleware/TestMiddleware.js
window.TestMiddleware = () => true;

// middleware/kernel.js (single registration place)
const routeMiddleware = {
  test: window.TestMiddleware,
  'profile-complete': window.ProfileCompleteMiddleware
};

Object.entries(routeMiddleware).forEach(([alias, handler]) => {
  if (typeof handler === 'function') AppRouter.middleware(alias, handler);
});

// Load middleware script in index.html:
// <script defer src="middleware/TestMiddleware.js"></script>
// <script defer src="middleware/kernel.js"></script>

// Apply middleware on a route:
AppRouter.get('/middleware-check', 'HomeController', 'index', {
  name: 'middleware.check',
  middleware: ['test']
});
// Frontend auth state helper
Auth.login({ name: 'Demo User', role: 'member' });
Auth.check(); // true/false
Auth.user();  // current user object
Auth.logout();

Security note: always enforce real auth/authorization in your backend APIs.

Controllers Pattern

Controllers extend Controller, load templates, and send data to views.

class HomeController extends Controller {
  async docs() {
    if (!this.view.templates['docs']) {
      await this.view.loadTemplate('docs', 'resources/views/docs.html');
    }
    this.view.updateElement('#app', 'docs', {
      title: 'Vanilla SPA Framework Documentation'
    });
  }
}
window.HomeController = HomeController;

Template Syntax (Variables, Conditions, Loops)

Variables

<h1>{{ title }}</h1>
<p>{{ user.name }}</p>

Laravel-inspired output: {{ ... }} escapes HTML by default. Use {!! ... !!} only for trusted HTML fragments.

<!-- Escaped output (safe default) -->
<p>{{ article.title }}</p>

<!-- Raw output (trusted content only) -->
<div>{!! article.renderedHtml !!}</div>

Conditionals

  <div class="admin-panel">Admin Content</div>
@else
  <div class="basic-panel">Basic Content</div>
@endif

@endunless

@isset(user.name)

@endisset

@empty(users)

@endempty

@auth

Conditional Attributes + Events

<input type="checkbox" @checked(user.active) />
<option @selected(user.role == 'admin')>Admin</option>
<button @disabled(formBusy) @required(needsConfirmation)>Save</button>

<div @class({ 'card': true, 'card--active': user.active ? true : false })>...</div>
<div @style({ color: user.color, display: isHidden ? 'none' : 'block' })>...</div>

<!-- Generic event directive -->
<button @on('click', 'handleSave(event)')>Save</button>

<!-- Shorthand event directives -->
<input @onchange('handleChange(event)') />
<button @onclick('handleClick(event)')>Click</button>

Ternary expressions are supported in template expressions, for example {{ user.isAdmin ? 'Admin' : 'Member' }}.

Null-coalescing and Elvis are supported too: {{ user.nickname ?? user.name }} and {{ user.nickname ?: user.name }}.

Custom Directives

// Register once during app boot:
window.View.directive('uppercase', (expression, data, view) => {
  const value = view.resolveExpressionValue(expression, data);
  return String(value || '').toUpperCase();
});

// Use in template:
@uppercase('hello')
@uppercase(user.name)

Use one API helper for internal controller routes and external network APIs.

// In controller methods
const users = await this.api.get('/api/users');

// DOM helper methods
this.api.updateElement('#users', '/api/users');

Form Handling

Add data attributes and let the framework submit/dispatch events automatically.

<form data-action="/api/users" data-method="POST" data-target="#result">
  <input type="text" name="name" required />
  <input type="email" name="email" required />
  <button type="submit">Submit</button>
</form>
<div id="result"></div>

You can listen to form-success and form-error events on the form.

Validation (Laravel-like Rules)

Use data-validate on forms to run client-side validation before API calls.

<form
  data-action="/api/users"
  data-method="POST"
  data-target="#result"
  data-validate='{"name":"required|min:3|max:40","email":"required|email"}'
  data-validate-messages='{"name.required":"Please enter your name"}'
>
  <input name="name" />
  <input name="email" />
  <button type="submit">Submit</button>
</form>
<div id="result"></div>
// Programmatic validation
const result = Validator.validate(
  { email: 'john@example.com' },
  { email: 'required|email' }
);

if (result.fails) {
  console.log(result.errors);
}
// FormRequest-style validation
class DemoSignupRequest extends FormRequest {
  rules() {
    return {
      name: 'required|min:3',
      email: 'required|email'
    };
  }

  messages() {
    return {
      'name.required': 'Name is required.'
    };
  }
}
window.DemoSignupRequest = DemoSignupRequest;

// In HTML form: data-request="DemoSignupRequest"
// Blade-style validation message block
@error('email')
  <small class="field-error">{{ error }}</small>
@enderror

// Named bag variant
@error('email', 'login')
  <small class="field-error">{{ error }}</small>
@enderror
// old input helper
<input name="email" value="{{ old('email', user.email ?? '') }}" />
// Directive form:
@old('email', user.email)

Live Validation Demo

Set data-validate-live="true" to validate while users type and auto-highlight fields.

<form
  data-action="/api/users"
  data-method="POST"
  data-target="#docs-validation-result"
  data-validate-live="true"
  data-validate-debounce="250"
  data-validate='{"name":"required|min:3","email":"required|email"}'
>
  <input name="name" placeholder="Name" />
  <input name="email" placeholder="Email" />
</form>

Response

Field-level errors render below inputs. Use data-validate-debounce (ms) to reduce noisy validation while typing. Supported rules: required, email, min, max, numeric, url, in.
